# tropek/ui Refactoring Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Full rewrite of `tropek/ui/src/` applying feature-sliced architecture, shadcn/ui, React Hook Form, and MSW — replacing ~1600 lines of god components and scattered duplication with a clean, layered codebase.

**Architecture:** Feature-sliced design (`features/evaluations/`, `features/assets/`, `features/slos/`). Each feature owns its types, API functions, custom hooks, and components. Pages are thin shells (~40–50 lines) that compose feature components. Shared utilities live in `lib/`. MSW intercepts all `/api/*` requests in development — fetch functions are always pure.

**Tech Stack:** React 19, TypeScript 5.9, Vite 8, TanStack React Query v5, React Router v7, Tailwind v3, shadcn/ui, React Hook Form + Zod, MSW v2, ECharts 6, Vitest 4

```typescript
// src/lib/format.ts
// Shared number and date formatters.
// Previously redefined inline in SLIBreakdownTable, EvaluationDetailPage,
// EvaluationListPage, and EvaluationHeatmap.

export function fmt(v: number | null | undefined): string {
  if (v == null) return '—'
  return v.toFixed(2)
}

export function fmtPct(v: number | null | undefined): string {
  if (v == null) return '—'
  return `${v.toFixed(1)}%`
}

export function fmtSlot(slot: string): string {
  const d = new Date(slot)
  if (isNaN(d.getTime())) return slot
  return d.toISOString().slice(11, 16) // "HH:MM"
}

export function fmtDate(iso: string): string {
  if (!iso) return '—'
  return iso.slice(0, 10) // "YYYY-MM-DD"
}
```

```typescript
// Centralised React Query key factory.
// Eliminates magic-string query keys scattered across components.
// Usage: useQuery({ queryKey: evaluationKeys.list(filters), queryFn: ... })
// EvaluationFilters canonical definition is in features/evaluations/types.ts.
// lib/ must not import from features/, so we use an inline structural type here.

type EvalFilters = { lab?: string; date?: string; slot?: string }

export const evaluationKeys = {
  all: ['evaluations'] as const,
  list: (filters: EvalFilters) => [...evaluationKeys.all, filters] as const,
  detail: (id: string) => [...evaluationKeys.all, id] as const,
  trend: (id: string, metric: string) => [...evaluationKeys.detail(id), metric] as const,
}

export const assetKeys = {
  all: ['assets'] as const,
  groups: () => [...assetKeys.all, 'groups'] as const,
}

export const sloKeys = {
  all: ['slos'] as const,
  detail: (name: string) => [...sloKeys.all, name] as const,
```

```typescript
// ---------------------------------------------------------------------------
// MSW-compatible wrappers
// Simple signatures for use in MSW handlers. Cache the full list once per
// module load (deterministic generator produces the same output every time).
// ---------------------------------------------------------------------------

let _cached: EvaluationSummary[] | null = null

function allEvals(): EvaluationSummary[] {
  if (!_cached) _cached = generateAllEvaluations()
  return _cached
}

export interface EvaluationListFilters {
  lab?: string
  date?: string
  slot?: string
}

export function getEvaluations(filters: EvaluationListFilters = {}): EvaluationSummary[] {
  let evals = allEvals()
  if (filters.lab) evals = evals.filter(e => e.asset_snapshot.tags?.['lab'] === filters.lab)
  if (filters.date) evals = evals.filter(e => e.start.startsWith(filters.date!))
  if (filters.slot) evals = evals.filter(e => e.start === filters.slot)
  return evals
}

export function getEvaluationDetail(id: string): EvaluationDetail {
  return generateEvaluationDetail(id, allEvals())
}

export function getTrend(evalId: string, metric: string): TrendPoint[] {
  // evalId format from generator: "{assetName}-{testName}-day{N}-slot{N}"
  const ev = allEvals().find(e => e.id === evalId)
  if (!ev) return []
  return generateTrendData(ev.name, metric, ev.asset_snapshot.name, allEvals())
}

export function getAssetGroupTree() {

  const assets = (assetsFixture as { items: Array<{ id: string; name: string; labels: Record<string, string> }> }).items
  const labMap = new Map<string, typeof assets>()
  for (const a of assets) {
    const lab = a.labels['lab'] ?? 'unknown'
    if (!labMap.has(lab)) labMap.set(lab, [])
    labMap.get(lab)!.push(a)
  }

    id: `group-${i}`,
    name: lab,
    display_name: lab.replace(/-/g, ' ').replace(/\b\w/g, c => c.toUpperCase()),
    members: members.map(a => ({ asset_id: a.id, asset_name: a.name, weight: 1 })),
    subgroups: [],
  }))

  return { top_level: allGroups, all_groups: allGroups }
}
```

```typescript
  return import('../generate')

export const evaluationHandlers = [
  http.get('/api/evaluations', async ({ request }) => {
    const url = new URL(request.url)
    const lab = url.searchParams.get('lab') ?? undefined
    const date = url.searchParams.get('date') ?? undefined
    const slot = url.searchParams.get('slot') ?? undefined
    const { getEvaluations } = await gen()
    return HttpResponse.json(getEvaluations({ lab, date, slot }))
  }),

  http.get('/api/evaluations/:id', async ({ params }) => {
    const { getEvaluationDetail } = await gen()
    return HttpResponse.json(getEvaluationDetail(params.id as string))
  }),

  http.get('/api/trend', async ({ request }) => {
    const url = new URL(request.url)
    const evalId = url.searchParams.get('eval_id') ?? ''
    const metric = url.searchParams.get('metric') ?? ''
    const { getTrend } = await gen()
    return HttpResponse.json(getTrend(evalId, metric))
  }),

  http.post('/api/evaluations', async ({ request }) => {
    const body = await request.json() as Record<string, unknown>
    // Return a stub triggered evaluation response.
    return HttpResponse.json(
      { id: crypto.randomUUID(), status: 'triggered', ...body },
      { status: 202 }
    )
  }),

  http.post('/api/evaluations/:id/annotations', async ({ params, request }) => {
    const body = await request.json() as Record<string, unknown>
    return HttpResponse.json(
      {
        id: crypto.randomUUID(),
        eval_id: params.id,
        created_at: new Date().toISOString(),
        ...body,
      },
      { status: 201 }
    )
  }),

  http.patch('/api/evaluations/:id/invalidate', async ({ params }) => {
    return HttpResponse.json({ id: params.id, invalidated: true })
  }),
```

```typescript
/**
 * Pure helper — exported so it can be unit-tested without React.
 * Guards required columns and toggles membership in the visible set.
 */
export function toggleColumnKey(
  prev: Set<string>,
  key: string,
  allCols: ColumnDef[]
): Set<string> {
  const col = allCols.find(c => c.key === key)
  if (col?.required) return prev
  const next = new Set(prev)
  next.has(key) ? next.delete(key) : next.add(key)
  return next
}

export function useColumnVisibility(dynamicCols: ColumnDef[]) {
  const allCols = [...FIXED_COLS, ...dynamicCols]
  const [visibleKeys, setVisibleKeys] = useState<Set<string>>(
    new Set(FIXED_COLS.filter(c => c.required).map(c => c.key))
  )
  const [open, setOpen] = useState(false)
  const pickerRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    function handleClick(e: MouseEvent) {
      if (pickerRef.current && !pickerRef.current.contains(e.target as Node)) {
        setOpen(false)
      }
    }
    if (open) document.addEventListener('mousedown', handleClick)
    return () => document.removeEventListener('mousedown', handleClick)
  }, [open])

  function toggle(key: string) {
    setVisibleKeys(prev => toggleColumnKey(prev, key, allCols))
  }

  return { allCols, visibleKeys, open, setOpen, toggle, pickerRef }
}
```

```typescript
import ReactECharts from 'echarts-for-react'
import { useMemo } from 'react'
import { RESULT_COLOUR } from '../constants'
import { fmtSlot } from '@/lib/format'
import type { EvaluationSummary } from '../types'

interface Props {
  evaluations: EvaluationSummary[]
  selectedDate: string | null
  onDateSelect: (date: string | null) => void
}

type ResultKey = keyof typeof RESULT_COLOUR

const RESULT_RANK: Record<string, number> = { error: 3, fail: 2, warning: 1, pass: 0 }

// Aggregate evaluations into heatmap cells (one cell per asset×test×slot).
// If multiple evaluations share a slot, take the worst result and average score.
function buildHeatmapData(evals: EvaluationSummary[]) {
  const slots = Array.from(new Set(evals.map(e => e.start))).sort()
  const rows = Array.from(
    new Set(evals.map(e => `${e.asset_snapshot.name} · ${e.name}`))
  ).sort()
  const cellMap = new Map<string, { result: string; score: number; count: number }>()

  for (const e of evals) {
    const rowKey = `${e.asset_snapshot.name} · ${e.name}`
    const colKey = e.start
    const key = `${rowKey}::${colKey}`
    const existing = cellMap.get(key)
    if (!existing) {
      cellMap.set(key, { result: e.result, score: e.score, count: 1 })
    } else {
      const rank = (r: string) => RESULT_RANK[r] ?? 0
      cellMap.set(key, {
        result: rank(e.result) > rank(existing.result) ? e.result : existing.result,
        score: (existing.score * existing.count + e.score) / (existing.count + 1),
        count: existing.count + 1,
      })
    }
  }

  const data: [number, number, number, string][] = []
  for (const [key, cell] of cellMap) {
    const [rowKey, colKey] = key.split('::')
    const xi = slots.indexOf(colKey)
    const yi = rows.indexOf(rowKey)
    data.push([xi, yi, Math.round(cell.score), cell.result])
  }
  return { slots, rows, data }
}

export function EvaluationHeatmap({ evaluations, selectedDate, onDateSelect }: Props) {
  const { slots, rows, data } = useMemo(() => buildHeatmapData(evaluations), [evaluations])

  const option = {
    renderer: 'svg',
    tooltip: {
      formatter: (p: { data: [number, number, number, string] }) => {
        const [xi, yi, score, result] = p.data
        return `${rows[yi]}<br/>${fmtSlot(slots[xi])}<br/>Score: ${score} · ${result}`
      },
    },
    xAxis: {
      type: 'category',
      data: slots.map(fmtSlot),
      axisLabel: { rotate: 45, fontSize: 11 },
    },
    yAxis: { type: 'category', data: rows, axisLabel: { fontSize: 11 } },
    visualMap: { show: false, min: 0, max: 100 },
    series: [{
      type: 'heatmap',
      data,
      itemStyle: {
        color: (p: { data: [number, number, number, string] }) =>
          RESULT_COLOUR[p.data[3] as ResultKey] ?? '#444',
      },
      emphasis: { itemStyle: { shadowBlur: 6, shadowColor: 'rgba(0,0,0,0.5)' } },
    }],
    grid: { top: 20, bottom: 80, left: 200, right: 20 },
  }

  return (
    <div className="w-full">
      <ReactECharts
        option={option}
        style={{ height: Math.max(200, rows.length * 28 + 100) }}
        onEvents={{
          click: (p: { data: [number, number, number, string] }) => {
            const slot = slots[p.data[0]]
            onDateSelect(selectedDate === slot ? null : slot)
          },
        }}
      />
    </div>
  )
}
```

```typescript
// src/features/assets/utils.ts
// Shared asset utility functions.
// Fixes DRY violation: collectGroupAssets() was duplicated in api/client.ts
// and TriggerEvaluationModal.tsx. Single implementation here.
// NOTE: Function renamed from `collectGroupAssets()` (spec name) to
// `collectGroupAssetNames()` to make the return type explicit in the name.

import type { AssetGroup, AssetGroupTree } from './types'

/**
 * Recursively collect all asset names from a group and its subgroups.
 * Python analogy: a recursive generator function over a tree structure.
 */
export function collectGroupAssetNames(
  group: AssetGroup,
  tree: AssetGroupTree
): string[] {
  const directMembers = group.members.map(m => m.asset_name)
  const subgroupMembers = group.subgroups.flatMap(sg => {
    const subGroup = tree.all_groups.find(g => g.id === sg.group_id)
    return subGroup ? collectGroupAssetNames(subGroup, tree) : []
  })
  return [...new Set([...directMembers, ...subgroupMembers])]
}
```

```typescript
/**
 * Serialises form values to a minimal YAML string that the backend validate
 * endpoint can parse. Includes spec_version, objectives block, and preserves
 * score_thresholds from the original SLO when present.
 */
function buildYamlFromObjectives(slo: SloDefinition, objectives: FormValues['objectives']): string {
  const lines: string[] = [
    'spec_version: "1.0"',
    `display_name: "${slo.display_name ?? slo.name}"`,
    'objectives:',
    ...objectives.flatMap(obj => [
      `  - sli: ${obj.sli}`,
      `    pass:`,
      `      - criteria: [${obj.pass_criteria}]`,
      ...(obj.warning_criteria
        ? ['    warning:', `      - criteria: [${obj.warning_criteria}]`]
        : []),
      `    weight: ${obj.weight}`,
      `    key_sli: ${obj.key_sli}`,
    ]),
  ]
  if (slo.score_thresholds) {
    lines.push('total_score:')
    if (slo.score_thresholds.total_pass != null) lines.push(`  pass: ${slo.score_thresholds.total_pass}`)
    if (slo.score_thresholds.total_warning != null) lines.push(`  warning: ${slo.score_thresholds.total_warning}`)
  }
  return lines.join('\n')
}
```

```typescript
  const [dragOver, setDragOver] = useState(false)
  const inputRef = useRef<HTMLInputElement>(null)
  const validation = useSloValidation()

  function readFile(file: File) {
    const reader = new FileReader()
    reader.onload = e => {
      const text = e.target?.result as string
      setYaml(text)
      validation.reset()
    }
    reader.readAsText(file)
  }

  function handleDrop(e: React.DragEvent) {
    e.preventDefault()
    setDragOver(false)
    const file = e.dataTransfer.files[0]
    if (file) readFile(file)
  }

  function handleValidate() {
    if (!yaml) return
    validation.mutate(yaml)
  }

  return (
    <div className="space-y-4">
      {/* Drop zone */}
      <div
        onDragOver={e => { e.preventDefault(); setDragOver(true) }}
        onDragLeave={() => setDragOver(false)}
        onDrop={handleDrop}
        onClick={() => inputRef.current?.click()}
        className={`border-2 border-dashed rounded-lg p-8 text-center cursor-pointer transition-colors ${
          dragOver ? 'border-blue-400 bg-blue-400/10' : 'border-gray-600 hover:border-gray-400'
        }`}
      >
        <input
          ref={inputRef}
          type="file"
          accept=".yaml,.yml"
          className="hidden"
          onChange={e => { const f = e.target.files?.[0]; if (f) readFile(f) }}
```

```typescript
import { useState } from 'react'
```

```typescript
type Mode = 'view' | 'edit-rows' | 'edit-yaml' | 'upload'

// SloDetail is a self-contained sub-component: mode state lives entirely here.
// It is not exported — only used by SloRegistryPage below.
function SloDetail({ name }: { name: string }) {
  const { data: slo, isLoading, isError } = useSloDetail(name)
  const [mode, setMode] = useState<Mode>('view')

  if (isLoading) return <p className="text-gray-400 text-sm py-2">Loading...</p>
  if (isError || !slo) return <p className="text-sm py-2">Failed to load data.</p>

  return (
    <div className="border-t border-gray-700 mt-2 pt-4 space-y-4">
      <div className="flex gap-2">
        <Button size="sm" variant="outline" onClick={() => setMode('upload')}>Upload YAML</Button>
        <Button size="sm" variant="outline" onClick={() => setMode('edit-rows')}>Edit Rows</Button>
        <Button size="sm" variant="outline" onClick={() => setMode('edit-yaml')}>Raw Edit</Button>
        <Button size="sm" variant="outline" disabled title="Coming in a future release">Test SLO</Button>
      </div>

      {mode === 'view' && (
        <>
          <SloObjectiveTable slo={slo} />
          {slo.slo_yaml && <SloYamlViewer yaml={slo.slo_yaml} />}
        </>
      )}
      {mode === 'edit-rows' && (
        <SloObjectiveEditor slo={slo} onCancel={() => setMode('view')} onSaved={() => setMode('view')} />
      )}
      {mode === 'edit-yaml' && (
        <SloYamlEditor slo={slo} onCancel={() => setMode('view')} onSaved={() => setMode('view')} />
      )}
      {mode === 'upload' && (
        <SloYamlUpload onCancel={() => setMode('view')} onSaved={() => setMode('view')} />
      )}
    </div>
  )
}

export function SloRegistryPage() {
  const { data: slos, isLoading, isError } = useSlos()
  const [expandedSlo, setExpandedSlo] = useState<string | null>(null)

  return (

      <h1 className="text-xl font-semibold">SLO Registry</h1>
      {slos.map(slo => (
        <div key={slo.name} className="border border-gray-700 rounded-lg p-4">

            onClick={() => setExpandedSlo(prev => prev === slo.name ? null : slo.name)}
          >
            <div className="flex items-center gap-3">
              <span className="font-semibold">{slo.display_name ?? slo.name}</span>
              <span className="text-xs text-gray-500">v{slo.version}</span>
              {slo.active && <Badge className="bg-green-700 text-green-100 text-xs">active</Badge>}
            </div>
            <span className="text-gray-500">{expandedSlo === slo.name ? '▼' : '▶'}</span>
          </div>
          {expandedSlo === slo.name && (
            <SloDetail name={slo.name} />
          )}
        </div>
      ))}
    </div>
  )
}
```
